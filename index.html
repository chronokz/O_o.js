Redirects URL, Domain, Hash

Get Domain-name

// The variable "domainName" value will be equal current domain name
var domainName = O_o.domain();

Get URL-address & Redirect

// Get current URL:
var domainName = O_o.url();
// Redirect to Wikipedia
O_o.url("http://www.wikipedia.org");

Get & Set Hash

// For example get hash #var1=test1&var2=test2
var hash = O_o.hash();
//This will return value:
var hash = "var1=test1&var2=test2";
 
// For example set hash
O_o.hash('new_any_hash');

Parse paramters from Hash

// The settings should be passed to Get-params. For example #var1=test1&var2=test2
var obj = O_o.hashParams();
//This will return object:
var obj = {
	"var1": "test1",
	"var2": "test2"
};

Get data of Date from Date-object / DATE / DATETIME


    
    

      
Allow paramters: 
           
          Day
          $d - Day of the month, 2 digits with leading zeros
          $j - Day of the month without leading zeros
           
          Month
          $m - Numeric representation of a month, with leading zeros
          $n - Numeric representation of a month, without leading zeros
          $F - A full textual representation of a month, such as January or March
          $M - A short textual representation of a month, three letters
           
          Year
          $Y - A full numeric representation of a year, 4 digits
          $y - A two digit representation of a year
           
          Hour
          $H - 24-hour format of an hour with leading zeros
          $G - 24-hour format of an hour without leading zeros
          $h - 12-hour format of an hour with leading zeros
          $g - 12-hour format of an hour without leading zeros
           
          Part of Day
          $a - Lowercase Ante meridiem and Post meridiem
          $A - Uppercase Ante meridiem and Post meridiem
           
          Minutes
          $i - Minutes with leading zeros
           
          Seconds
          $s - Seconds, with leading zeros

    

  

  

    
// Several examples of usage
O_o.date('2014-03-31 10:13:05', '$H:$i | $M $d, $Y'); // 10:13 | Mar 31, 2014
O_o.date(new Date(), '$Y-$m-$d $H:$i:$s');            // 2014-03-31 10:13:05

JSON

  

  
Cross-browser JSON.parse()

  

    
// Use like JSON.parse()
O_o.parseJson('{}');              // {}
O_o.parseJson('true');            // true
O_o.parseJson('"foo"');           // "foo"
O_o.parseJson('[1, 5, "false"]'); // [1, 5, "false"]
O_o.parseJson('null');            // null


  

  
Cross-browser JSON.stringify()

  

    
// Use like JSON.stringify()
O_o.stringJson({});                  // '{}'
O_o.stringJson(true);                // 'true'
O_o.stringJson("foo");               // '"foo"'
O_o.stringJson([1, "false", false]); // '[1,"false",false]'
O_o.stringJson({ x: 5 });            // '{"x":5}'
O_o.stringJson({x: 5, y: 6});        // '{"x":5,"y":6}' or '{"y":6,"x":5}'
